Code Snippet:

// call an api function to add GeoJson layer
const layerID = cgpv.api.map('LYR4').createGeoJSONLayer({ {
type: 'geoJSON',
url: 'https://raw.githubusercontent.com/fgpv-vpgf/contributed-plugins/master/chart/src/samples/geoJSON/pieChart.json',
});

// removing a geojson layer using the ID
cgpv.api.map('LYR4').remove(document.getElementById('new-layer-id').innerText);
